~~~
Algoritmo "ConteAte?"

Var
   valor,salto,contador: inteiro
Inicio
      contador <- 0
      Escreva("Quer conta ate quanto? ")
      Leia(valor)
      Escreva("O numero ira saltar de quanto em quanto? ")
      Leia(salto)
      Enquanto (contador <= valor) faca
           EscrevaL(contador)
           contador <- contador + salto
      FimEnquanto


Fimalgoritmo
